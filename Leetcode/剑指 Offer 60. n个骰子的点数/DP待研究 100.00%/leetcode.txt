class Solution {
    public double[] dicesProbability(int n) {
        double[] pre={1/6d,1/6d,1/6d,1/6d,1/6d,1/6d};
        for(int i=2;i<=n;++i){
            double[] res=new double[(5*i)+1];
            for(int j=0;j<pre.length;++j){
                for(int k=0;k<6;++k){
                    res[j+k]+=pre[j]/6;
                }
            }
            pre=res;
        }
        return pre;
    }
}
